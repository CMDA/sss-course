What is a route?

URL

URL

Route


app.get('/hello/amsterdam', function(req, res){
  res.locals.location = "Amsterdam";
  res.render('hello');
});

Route + Query string


app.get('/hello', function(req, res){
  res.locals.location = req.query.location;
  res.render('hello');
});
http://expressjs.com/4x/api.html#req.query

Route + Parameter


app.get('/hello/:location', function(req, res){
  res.locals.location = req.params.location;
  res.render('hello');
});

Router module


// routes/hello.js
var express = require('express');
var router = express.Router();

router.get('/hello', function(req, res){
  res.send('... you say goodbye, and I say hello!');
});

module.exports = router;

Using the router


// app.js
var helloRouter = require('./routes/hello');

app.use('/hello', helloRouter);

Go to ...

http://localhost:3000/hello/hello

What happened?

  1. Start with the app
  2. /hello/hello → app.use('/hello', helloRouter)
  3. Continue in router
  4. /hello/hello → router.get('/hello', ...)
