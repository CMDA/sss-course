What is a route?

A function that responds to part of an URL

URL

URL

Route


app.get('/hello/amsterdam', function(req, res){
  res.locals.name = "You";
  res.locals.location = "Amsterdam";
  res.render('hello');
});

Route + Parameter


app.get('/hello/:location', function(req, res){
  res.locals.name = "You";
  res.locals.location = req.params.location;
  res.render('hello');
});

Query string


  // routes/weather.js
  var express = require('express');
  var router = express.Router();

  router.get('/', function(req, res){
    var location = req.query.location;
    res.send('Who is the weather in ' + location + '?');
  });

  module.exports = router;

Routers

Router

  • URL structure
  • HTTP request handling
  • HTTP response generation

Router example


// routes/weather.js
var express = require('express');
var router = express.Router();

router.get('/', function(req, res){
  res.send('Hello world (from router)!');
});

module.exports = router;

Using the router


// app.js
var express = require('express');
var app = express();
var weatherRouter = require('./routes/weather');

app.use('/weather', weatherRouter);

// Start the server
app.listen(3000);

Using multiple routers


// app.js
var express = require('express');
var app = express();
var weatherRouter = require('./routes/weather');
var indexRouter = require('./routes/index');

app.use('/', indexRouter);
app.use('/weather', weatherRouter);

// Start the server
app.listen(3000);

Consuming APIs

Request

https://www.npmjs.org/package/request

Installing Request


$ npm install --save request

Example


var request = require("request");
var url = "http://api.openweathermap.org/data/2.5/weather?q=Amsterdam,nl";
request(url, function(error, response, body){
  if (!error && response.statusCode == 200) {
    console.log(body) // Print the weather.
  }
});

var calculate = function(expenses){
  var calculate = {};
  expenses.forEach(function(expense){
    if(calculate[expense.type]){
      calculate[expense.type] += expense.kosten;
    } else {
      calculate[expense.type] = expense.kosten;
    }
  });

  return calculate;
};
